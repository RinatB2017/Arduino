295 size_t Print::printNumber(unsigned long n, uint8_t base) {
296  char buf[8 * sizeof(long) + 1]; // Assumes 8-bit chars plus zero byte.
297  char *str = &buf[sizeof(buf) - 1];
298 
299  *str = '\0';
300 
301  // prevent crash if called with base == 1
302  if (base < 2) base = 10;
303 
304  do {
305  unsigned long m = n;
306  n /= base;
307  char c = m - base * n;
308  *--str = c < 10 ? c + '0' : c + 'A' - 10;
309  } while(n);
310 
311  return write(str);
312 }

314 size_t Print::printFloat(double number, uint8_t digits)
315 {
316  size_t n = 0;
317 
318  if (isnan(number)) return print("nan");
319  if (isinf(number)) return print("inf");
320  if (number > 4294967040.0) return print ("ovf"); // constant determined empirically
321  if (number <-4294967040.0) return print ("ovf"); // constant determined empirically
322 
323  // Handle negative numbers
324  if (number < 0.0)
325  {
326  n += print('-');
327  number = -number;
328  }
329 
330  // Round correctly so that print(1.999, 2) prints as "2.00"
331  double rounding = 0.5;
332  for (uint8_t i=0; i<digits; ++i)
333  rounding /= 10.0;
334 
335  number += rounding;
336 
337  // Extract the integer part of the number and print it
338  unsigned long int_part = (unsigned long)number;
339  double remainder = number - (double)int_part;
340  n += print(int_part);
341 
342  // Print the decimal point, but only if there are digits beyond
343  if (digits > 0) {
344  n += print(".");
345  }
346 
347  // Extract digits from the remainder one at a time
348  while (digits-- > 0)
349  {
350  remainder *= 10.0;
351  int toPrint = int(remainder);
352  n += print(toPrint);
353  remainder -= toPrint;
354  }
355 
356  return n;
357 }
