12 /*
13  Debounce
14 
15  Each time the input pin goes from LOW to HIGH (e.g. because of a push-button
16  press), the output pin is toggled from LOW to HIGH or HIGH to LOW. There's
17  a minimum delay between toggles to debounce the circuit (i.e. to ignore
18  noise).
19 
20  The circuit:
21  * LED attached from pin 13 to ground
22  * pushbutton attached from pin 9 to +5V
23  * 10K resistor attached from pin 9 to ground
24 
25  * Note: On most Arduino boards, there is already an LED on the board
26  connected to pin 13, so you don't need any extra components for this example.
27 
28 
29  created 21 November 2006
30  by David A. Mellis
31  modified 30 Aug 2011
32  by Limor Fried
33  modified 28 Dec 2012
34  by Mike Walters
35 
36  This example code is in the public domain.
37 
38  http://www.arduino.cc/en/Tutorial/Debounce
39  */

41 // constants won't change. They're used here to
42 // set pin numbers:
43 const int buttonPin = 9; // the number of the pushbutton pin
44 const int ledPin = 13; // the number of the LED pin
45 
46 // Variables will change:
47 int ledState = HIGH; // the current state of the output pin
48 int buttonState; // the current reading from the input pin
49 int lastButtonState = LOW; // the previous reading from the input pin
50 
51 // the following variables are long's because the time, measured in miliseconds,
52 // will quickly become a bigger number than can be stored in an int.
53 long lastDebounceTime = 0; // the last time the output pin was toggled
54 long debounceDelay = 50; // the debounce time; increase if the output flickers
55 
56 void setup() {
57  pinMode(buttonPin, INPUT);
58  pinMode(ledPin, OUTPUT);
59 
60  // set initial LED state
61  digitalWrite(ledPin, ledState);
62 }

64 void loop() {
65  // read the state of the switch into a local variable:
66  int reading = digitalRead(buttonPin);
67 
68  // check to see if you just pressed the button
69  // (i.e. the input went from LOW to HIGH), and you've waited
70  // long enough since the last press to ignore any noise:
71 
72  // If the switch changed, due to noise or pressing:
73  if (reading != lastButtonState) {
74  // reset the debouncing timer
75  lastDebounceTime = millis();
76  }
77 
78  if ((millis() - lastDebounceTime) > debounceDelay) {
79  // whatever the reading is at, it's been there for longer
80  // than the debounce delay, so take it as the actual current state:
81 
82  // if the button state has changed:
83  if (reading != buttonState) {
84  buttonState = reading;
85 
86  // only toggle the LED if the new button state is HIGH
87  if (buttonState == HIGH) {
88  ledState = !ledState;
89  }
90  }
91  }
92 
93  // set the LED:
94  digitalWrite(ledPin, ledState);
95 
96  // save the reading. Next time through the loop,
97  // it'll be the lastButtonState:
98  lastButtonState = reading;
99 }
